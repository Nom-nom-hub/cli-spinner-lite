cli-spinner-lite

The ultimate zero-dependency utility for Node.js CLI tools. Elevate your command-line experience with stunning, animated spinners during asynchronous operations. Lightweight, powerful, and infinitely customizable.

Key Features

  • Zero Dependencies: Keep your project lean and fast.
  • Highly Customizable: Control text, frames, intervals, and colors.
  • Dynamic Updates: Change spinner text on the fly.
  • Promise Integration: Seamlessly wrap async functions with `withSpinner`.
  • Cross-Platform: Works flawlessly on macOS, Windows, and Linux.
  • TypeScript Support: Full type definitions for modern development.

Installation

npm install cli-spinner-lite

Usage Examples (Node.js)

Basic Spinner with Dynamic Text and Color


const { Spinner } = require('cli-spinner-lite');
const spinnerStyles = require('cli-spinner-lite/spinnerStyles');

async function basicSpinnerDemo() {
  console.log("--- Basic Spinner Demo ---");
  const spinner = new Spinner({
    text: 'Initializing core modules...',
    frames: spinnerStyles.dots.frames,
    interval: spinnerStyles.dots.interval,
    color: 'blue' // Customizable color
  });
  spinner.start();
  await new Promise(resolve => setTimeout(resolve, 2500));
  spinner.updateText('Fetching remote configurations...'); // Dynamic text update
  await new Promise(resolve => setTimeout(resolve, 2000));
  spinner.stop('✅ Core modules ready!'); // Stop with success message
}

basicSpinnerDemo();

Using withSpinner for Promise Integration and Error Handling


const { withSpinner } = require('cli-spinner-lite');
const spinnerStyles = require('cli-spinner-lite/spinnerStyles');

async function withSpinnerDemo() {
  console.log("\n--- withSpinner Utility Demo ---");
  try {
    await withSpinner(async () => {
      await new Promise(resolve => setTimeout(resolve, 3000));
      // Uncomment the line below to simulate an error:
      // throw new Error("Critical sub-process failed!");
    }, {
      text: "Executing complex algorithms",
      frames: spinnerStyles.line.frames,
      interval: spinnerStyles.line.interval,
      color: "green"
    });
  } catch (error) {
    console.error(`❌ Operation failed: ${error.message}`); // Error handling
  }
}

Chaining Multiple withSpinner Operations


const { withSpinner } = require('cli-spinner-lite');
const spinnerStyles = require('cli-spinner-lite/spinnerStyles');

async function chainedSpinnersDemo() {
  console.log("\n--- Chained Spinners Demo ---");
  await withSpinner(async () => { await new Promise((resolve) => setTimeout(resolve, 1200)); }, { text: "Task A: Data validation", frames: spinnerStyles.bounce.frames, interval: spinnerStyles.bounce.interval, color: "yellow" });
  await withSpinner(async () => { await new Promise((resolve) => setTimeout(resolve, 1200)); }, { text: "Task B: Database sync", frames: spinnerStyles.grow.frames, interval: spinnerStyles.grow.interval, color: "cyan" });
  await withSpinner(async () => { await new Promise((resolve) => setTimeout(resolve, 1200)); }, { text: "Task C: Report generation", frames: spinnerStyles.arc.frames, interval: spinnerStyles.arc.interval, color: "magenta" });
  console.log("✅ All chained tasks completed!");
}

Using a Custom Output Stream


const { Spinner } = require('cli-spinner-lite');
const spinnerStyles = require('cli-spinner-lite/spinnerStyles');

async function customStreamDemo() {
  console.log("\n--- Custom Stream Demo ---");
  // Create a mock stream for demonstration (can be any WritableStream)
  const mockStream = {
    _buffer: "",
    write(chunk) { this._buffer += chunk; },
    clearLine() { this._buffer = ""; },
    cursorTo() {},
    getOutput() { return this._buffer; }
  };
  const spinner = new Spinner({
    text: "Writing to mock stream",
    frames: spinnerStyles.circle.frames,
    interval: spinnerStyles.circle.interval,
    stream: mockStream, // Direct output to custom stream
    color: "red"
  });
  spinner.start();
  await new Promise(resolve => setTimeout(resolve, 2000));
  spinner.stop("✅ Mock stream operation done!");
  console.log("Mock stream content (last line):", mockStream.getOutput().split('\n').pop());
}

customStreamDemo();
